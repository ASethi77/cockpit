cmake_minimum_required(VERSION 3.5)

project(liveplot LANGUAGES CXX)

set(CMAKE_INCLUDE_CURRENT_DIR ON)

set(CMAKE_AUTOUIC ON)
set(CMAKE_AUTOMOC ON)
set(CMAKE_AUTORCC ON)

set(CMAKE_CXX_STANDARD 11)
set(CMAKE_CXX_STANDARD_REQUIRED ON)

find_package(Qt5 COMPONENTS Widgets PrintSupport REQUIRED)

set(LIVEPLOT_INCLUDE_DIR "${CMAKE_CURRENT_LIST_DIR}/include")

message(STATUS "includes are ${LIVEPLOT_INCLUDE_FILES}")

if(ANDROID)
  add_library(liveplot SHARED
    main.cpp
    liveplot.cpp
    liveplot.ui
  )
else()
  add_executable(liveplot
    main.cpp
    qcustomplot.cpp
    liveplot.cpp

    # Apparently you have to include headers like this in QtCreator's hosted CMake because
    # that's how it knows how to pull in headers used by autogenerated .ui components
    include/liveplot.h
    include/qcustomplot.h

    liveplot.ui
  )
endif()

target_include_directories(liveplot PRIVATE ${LIVEPLOT_INCLUDE_DIR})
target_link_libraries(liveplot PRIVATE Qt5::Widgets Qt5::PrintSupport)
